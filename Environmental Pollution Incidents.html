# url
url <- "https://environment.data.gov.uk/api/file/download?fileDataSetId=a5a11813-517a-478b-be89-4acdf2c1f105&fileName=Environmental%20Pollution%20Incidents%20(Category%201%20and%202).zip"

# download file
download.file(url, destfile = "data/pollution.zip", mode = "wb")

#unzip
unzip("data/pollution.zip",
      files = c("EP_Incidents_Nirs2.csv", "EP_Pollutants_Nirs2.csv"),
      exdir = "data")

# assign to variables
incidents <- rio::import("data/EP_Incidents_Nirs2.csv")
pollutants <- rio::import("data/EP_Pollutants_Nirs2.csv")

# date, year and month
incidents$NOT_DATE <- as.POSIXct(incidents$NOT_DATE, format = "%d/%m/%Y %H:%M")
incidents$YEAR <- as.numeric(format(incidents$NOT_DATE, "%Y"))
incidents$MONTH <- as.numeric(format(incidents$NOT_DATE, "%m"))

# Convert to spatial object (British National Grid)
df <- st_as_sf(incidents, coords = c("X_CONF", "Y_CONF"), crs = 27700)

# join datasets
df <- df %>%
  left_join(
    pollutants %>% select(NOT_ID, POLL_TYPE, POLLUTANT), 
    by = "NOT_ID")

# nominal categorical as factors
df$REGION_WM <- as.factor(df$REGION_WM)
df$AREA_WM <- as.factor(df$AREA_WM)
df$REGION_PF <- as.factor(df$REGION_PF)
df$AREA_PF <- as.factor(df$AREA_PF)
df$COUNTY <- as.factor(df$COUNTY)
df$UNITARY <- as.factor(df$UNITARY)
df$DISTRICT <- as.factor(df$DISTRICT)
df$EP_INC <- as.factor(df$EP_INC)
df$EIL_AIR <- as.factor(df$EIL_AIR)
df$EIL_LAND <- as.factor(df$EIL_LAND)
df$EIL_WATER <- as.factor(df$EIL_WATER)
df$POLL_TYPE <- as.factor(df$POLL_TYPE)
df$POLLUTANT <- as.factor(df$POLLUTANT)

# ordinal categorical as factors
df <- df %>%
  mutate(YEAR = factor(YEAR, levels = sort(unique(YEAR))))
df <- df %>%
  mutate(MONTH = factor(MONTH, levels = sort(unique(MONTH))))

# plot map, facet by year
ggplot(df) +
  geom_sf(aes(color = factor(EIL_WATER))) +
  facet_wrap(~ YEAR) +
  labs(
    x = "Longitude",
    y = "Latitude",
    color = "Environmental Impact Level (Water)",
    title = "Map of incidents by category (2000-2026)"
  )  +
  theme_bw() +
  theme(
    legend.position = "right"
  )
